void solve(int n) {
    map<int, vector<int>> mp;
    int maxD = 0;
    while(n--) {
        int p, d;
        cin >> p >> d;
        maxD = max(maxD, d);
        mp[d].push_back(p);
    }

    priority_queue<int, vector<int>, greater<int>> arr;
    for(auto it = mp.begin(); it != mp.end(); it++) {
        for(auto p : it->second) {
            if(arr.size() < it->first) {
                arr.push(p);
            } else if(arr.top() < p) {
                arr.pop();
                arr.push(p);
            }
        }
        
    }
    ll ans = 0;
    while(!arr.empty()) {
        ans += arr.top();
        arr.pop();
    }

    cout << ans << endl;
    
}

int main() {
    int n;
    while(cin >> n) {
        solve(n);
    }
    system("pause");
}
