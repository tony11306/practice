## 第4題：隔出最多的水
> 花費時間: 10分鐘，完成程度: 完全靠自己

經典雙指標面試題

```c++
class Solution {
public:
    int maxArea(vector<int>& height) {
        int l = 0;
        int r = height.size()-1;
        
        int ans = 0;
        
        while(l < r) {
            ans = max(ans, (r-l) * min(height[l], height[r]));
            if(height[l] < height[r]) {
                l++;
            } else {
                r--;
            }
        }
        return ans;
    }
};
```

## 第5題：跳到最後-V2
> 花費時間: 不知道，以前寫過的，完成程度: 完全靠自己
```c++
class Solution {
public:
    int jump(vector<int>& nums) {
        if(nums.size() == 1) {
            return 0;
        }
        int n = nums.size();
        int dp[n];
        memset(dp, 0x3f, sizeof(dp));
        dp[0] = 0;
        for(int i = 0; i < n; ++i) {
            for(int j = nums[i]; j >= 1; --j) {
                if(i + j < nums.size() - 1) {
                    dp[i+j] = min(dp[i+j], dp[i] + 1);
                } else {
                    return dp[i]+1;
                }
            }
        }
        
        return dp[n-1];
    }
};
```
